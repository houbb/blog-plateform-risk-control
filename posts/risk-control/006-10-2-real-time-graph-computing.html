        # è®¡ç®—æ¨¡å—åº¦
        modularity = 0.0
        for community in community_edges:
            internal_edges = community_edges[community]
            community_degree = community_degrees[community]

import hashlib

class GraphComputationCache:
    """å›¾è®¡ç®—ç¼“å­˜"""
    
    def __init__(self, max_size: int = 1000, ttl: int = 300):
        self.max_size = max_size
        self.ttl = ttl  # ç¼“å­˜è¿‡æœŸæ—¶é—´ï¼ˆç§’ï¼‰
        self.cache = OrderedDict()  # ä½¿ç”¨LRUç¼“å­˜ç­–ç•¥
        self.access_count = defaultdict(int)  # è®¿é—®è®¡æ•°
        self.hit_count = 0
        self.miss_count = 0
    
    def _generate_cache_key(self, func_name: str, args: tuple, kwargs: dict) -> str:
        """ç”Ÿæˆç¼“å­˜é”®"""
        # åºåˆ—åŒ–å‚æ•°
        key_data = f"{func_name}:{str(args)}:{str(sorted(kwargs.items()))}"
        # ä½¿ç”¨MD5ç”Ÿæˆå›ºå®šé•¿åº¦çš„é”®
        return hashlib.md5(key_data.encode()).hexdigest()
    
    def get(self, func_name: str, args: tuple, kwargs: dict) -> Optional[Any]:
        """è·å–ç¼“å­˜å€¼"""
        cache_key = self._generate_cache_key(func_name, args, kwargs)
        
        if cache_key in self.cache:
            entry = self.cache[cache_key]
            # æ£€æŸ¥æ˜¯å¦è¿‡æœŸ
            if time.time() - entry['timestamp'] < self.ttl:
                # æ›´æ–°è®¿é—®è®¡æ•°
                self.access_count[cache_key] += 1
                self.hit_count += 1
                
                # ç§»åŠ¨åˆ°æœ«å°¾ï¼ˆLRUï¼‰
                self.cache.move_to_end(cache_key)
                
                return entry['value']
            else:
                # è¿‡æœŸï¼Œåˆ é™¤ç¼“å­˜é¡¹
                del self.cache[cache_key]
                if cache_key in self.access_count:
                    del self.access_count[cache_key]
        
        self.miss_count += 1
        return None
    
    def put(self, func_name: str, args: tuple, kwargs: dict, value: Any):
        """æ”¾å…¥ç¼“å­˜å€¼"""
        cache_key = self._generate_cache_key(func_name, args, kwargs)
        
        # å¦‚æœç¼“å­˜å·²æ»¡ï¼Œåˆ é™¤æœ€å°‘ä½¿ç”¨çš„é¡¹
        if len(self.cache) >= self.max_size:
            # æ‰¾åˆ°è®¿é—®æ¬¡æ•°æœ€å°‘çš„é¡¹
            if self.access_count:
                min_access_key = min(self.access_count.keys(), 
                                   key=lambda k: self.access_count[k])
                del self.cache[min_access_key]
                del self.access_count[min_access_key]
            else:
                # å¦‚æœæ²¡æœ‰è®¿é—®è®¡æ•°ï¼Œåˆ é™¤æœ€è€çš„é¡¹
                self.cache.popitem(last=False)
        
        # æ·»åŠ æ–°é¡¹
        self.cache[cache_key] = {
            'value': value,
            'timestamp': time.time(),
            'func_name': func_name
        }
        self.access_count[cache_key] = 1
    
    def invalidate(self, func_name: Optional[str] = None):
        """ä½¿ç¼“å­˜å¤±æ•ˆ"""
        if func_name:
            # åªä½¿ç‰¹å®šå‡½æ•°çš„ç¼“å­˜å¤±æ•ˆ
            keys_to_remove = [
                key for key, entry in self.cache.items() 
                if entry['func_name'] == func_name
            ]
            for key in keys_to_remove:
                del self.cache[key]
                if key in self.access_count:
                    del self.access_count[key]
        else:
            # ä½¿æ‰€æœ‰ç¼“å­˜å¤±æ•ˆ
            self.cache.clear()
            self.access_count.clear()
    
    def get_stats(self) -> Dict[str, Any]:
        """è·å–ç¼“å­˜ç»Ÿè®¡ä¿¡æ¯"""
        total_requests = self.hit_count + self.miss_count
        hit_rate = self.hit_count / total_requests if total_requests > 0 else 0
        
        return {
            'hit_count': self.hit_count,
            'miss_count': self.miss_count,
            'hit_rate': hit_rate,
            'cache_size': len(self.cache),
            'max_size': self.max_size,
            'ttl': self.ttl
        }
    
    def get_hot_items(self, top_n: int = 10) -> List[Dict[str, Any]]:
        """è·å–çƒ­é—¨ç¼“å­˜é¡¹"""
        hot_items = []
        current_time = time.time()
        
        for cache_key, access_count in sorted(
            self.access_count.items(), 
            key=lambda x: x[1], 
            reverse=True
        )[:top_n]:
            if cache_key in self.cache:
                entry = self.cache[cache_key]
                hot_items.append({
                    'key': cache_key[:16] + '...',  # æˆªæ–­æ˜¾ç¤º
                    'func_name': entry['func_name'],
                    'access_count': access_count,
                    'age': current_time - entry['timestamp'],
                    'size': len(str(entry['value'])) if entry['value'] else 0
                })
        
        return hot_items

    """å›¾è®¡ç®—ç¼“å­˜è£…é¥°å™¨"""
    def decorator(func):

            cache.put(func.__name__, args, kwargs, result)
            
            return result
        return wrapper
    return decorator
